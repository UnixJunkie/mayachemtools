  92 def CalculateEnergy():
  93     """Calculate single point energy."""
  94     
  95     # Setup a molecule reader...
  96     MiscUtil.PrintInfo("\nProcessing file %s..." % OptionsInfo["Infile"])
  97     Mols  = RDKitUtil.ReadMolecules(OptionsInfo["Infile"], **OptionsInfo["InfileParams"])
  98     
  99     # Set up a molecule writer...
 100     Writer = RDKitUtil.MoleculesWriter(OptionsInfo["Outfile"], **OptionsInfo["OutfileParams"])
 101     if Writer is None:
 102         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % OptionsInfo["Outfile"])
 103     MiscUtil.PrintInfo("Generating file %s..." % OptionsInfo["Outfile"])
 104 
 105     MolCount, ValidMolCount, EnergyFailedCount = ProcessMolecules(Mols, Writer)
 106 
 107     if Writer is not None:
 108         Writer.close()
 109     
 110     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
 111     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
 112     MiscUtil.PrintInfo("Number of molecules failed during energy calculation: %d" % EnergyFailedCount)
 113     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + EnergyFailedCount))

 115 def ProcessMolecules(Mols, Writer):
 116     """Process and calculate energy of molecules."""
 117     
 118     if OptionsInfo["MPMode"]:
 119         return ProcessMoleculesUsingMultipleProcesses(Mols, Writer)
 120     else:
 121         return ProcessMoleculesUsingSingleProcess(Mols, Writer)
 122 
 123 def ProcessMoleculesUsingSingleProcess(Mols, Writer):
 124     """Process and calculate energy of molecules using a single process."""
 125     
 126     # Intialize Psi4...
 127     MiscUtil.PrintInfo("\nInitializing Psi4...")
 128     Psi4Handle = Psi4Util.InitializePsi4(Psi4RunParams = OptionsInfo["Psi4RunParams"], Psi4OptionsParams = OptionsInfo["Psi4OptionsParams"], PrintVersion = True, PrintHeader = True)
 129     OptionsInfo["psi4"] = Psi4Handle
 130 
 131     # Setup conversion factor for energy units...
 132     SetupEnergyConversionFactor(Psi4Handle)
 133     
 134     MiscUtil.PrintInfo("\nCalculating energy...")
 135     
 136     (MolCount, ValidMolCount, EnergyFailedCount) = [0] * 3
 137     for Mol in Mols:
 138         MolCount += 1
 139         
 140         if not CheckAndValidateMolecule(Mol, MolCount):
 141             continue
 142         
 143         # Setup a Psi4 molecule...
 144         Psi4Mol = SetupPsi4Mol(Psi4Handle, Mol, MolCount)
 145         if Psi4Mol is None:
 146             continue
 147         
 148         ValidMolCount += 1
 149         
 150         CalcStatus, Energy = CalculateMoleculeEnergy(Psi4Handle, Psi4Mol, Mol, MolCount)
 151 
 152         if not CalcStatus:
 153             if not OptionsInfo["QuietMode"]:
 154                 MiscUtil.PrintWarning("Failed to calculate energy for molecule %s" % RDKitUtil.GetMolName(Mol, MolCount))
 155             
 156             EnergyFailedCount += 1
 157             continue
 158         
 159         Energy = "%.*f" % (OptionsInfo["Precision"], Energy)
 160         WriteMolecule(Writer, Mol, Energy)
 161     
 162     return (MolCount, ValidMolCount, EnergyFailedCount)
 163 
 164 def ProcessMoleculesUsingMultipleProcesses(Mols, Writer):
 165     """Process and calculate energy of molecules using  process."""
 166     
 167     MiscUtil.PrintInfo("\nCalculating energy using multiprocessing...")
 168     
 169     MPParams = OptionsInfo["MPParams"]
 170     
 171     # Setup data for initializing a worker process...
 172     InitializeWorkerProcessArgs = (MiscUtil.ObjectToBase64EncodedString(Options), MiscUtil.ObjectToBase64EncodedString(OptionsInfo))
 173     
 174     # Setup a encoded mols data iterable for a worker process...
 175     WorkerProcessDataIterable = RDKitUtil.GenerateBase64EncodedMolStrings(Mols)
 176     
 177     # Setup process pool along with data initialization for each process...
 178     if not OptionsInfo["QuietMode"]:
 179         MiscUtil.PrintInfo("\nConfiguring multiprocessing using %s method..." % ("mp.Pool.imap()" if re.match("^Lazy$", MPParams["InputDataMode"], re.I) else "mp.Pool.map()"))
 180         MiscUtil.PrintInfo("NumProcesses: %s; InputDataMode: %s; ChunkSize: %s\n" % (MPParams["NumProcesses"], MPParams["InputDataMode"], ("automatic" if MPParams["ChunkSize"] is None else MPParams["ChunkSize"])))
 181     
 182     ProcessPool = mp.Pool(MPParams["NumProcesses"], InitializeWorkerProcess, InitializeWorkerProcessArgs)
 183     
 184     # Start processing...
 185     if re.match("^Lazy$", MPParams["InputDataMode"], re.I):
 186         Results = ProcessPool.imap(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
 187     elif re.match("^InMemory$", MPParams["InputDataMode"], re.I):
 188         Results = ProcessPool.map(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
 189     else:
 190         MiscUtil.PrintError("The value, %s, specified for \"--inputDataMode\" is not supported." % (MPParams["InputDataMode"]))
 191 
 192     # Print out Psi4 version in the main process...
 193     MiscUtil.PrintInfo("\nInitializing Psi4...\n")
 194     Psi4Handle  = Psi4Util.InitializePsi4(PrintVersion = True, PrintHeader = False)
 195     OptionsInfo["psi4"] = Psi4Handle
 196     
 197     (MolCount, ValidMolCount, EnergyFailedCount) = [0] * 3
 198     for Result in Results:
 199         MolCount += 1
 200         MolIndex, EncodedMol, CalcStatus, Energy = Result
 201         
 202         if EncodedMol is None:
 203             continue
 204         
 205         ValidMolCount += 1
 206 
 207         Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
 208         
 209         if not CalcStatus:
 210             if not OptionsInfo["QuietMode"]:
 211                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
 212                 MiscUtil.PrintWarning("Failed to calculate energy for molecule %s" % MolName)
 213             
 214             EnergyFailedCount += 1
 215             continue
 216         
 217         Energy = "%.*f" % (OptionsInfo["Precision"], Energy)
 218         WriteMolecule(Writer, Mol, Energy)
 219     
 220     return (MolCount, ValidMolCount, EnergyFailedCount)
 221 
 222 def InitializeWorkerProcess(*EncodedArgs):
 223     """Initialize data for a worker process."""
 224     
 225     global Options, OptionsInfo
 226     
 227     if not OptionsInfo["QuietMode"]:
 228         MiscUtil.PrintInfo("Starting process (PID: %s)..." % os.getpid())
 229     
 230     # Decode Options and OptionInfo...
 231     Options = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[0])
 232     OptionsInfo = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[1])
 233 
 234     # Psi4 is initialized in the worker process to avoid creation of redundant Psi4
 235     # output files for each process...
 236     OptionsInfo["Psi4Initialized"]  = False
 237 
 238 def InitializePsi4ForWorkerProcess():
 239     """Initialize Psi4 for a worker process."""
 240     
 241     if OptionsInfo["Psi4Initialized"]:
 242         return
 243 
 244     OptionsInfo["Psi4Initialized"] = True
 245     
 246     # Update output file...
 247     OptionsInfo["Psi4RunParams"]["OutputFile"] = Psi4Util.UpdatePsi4OutputFileUsingPID(OptionsInfo["Psi4RunParams"]["OutputFile"], os.getpid())
 248     
 249     # Intialize Psi4...
 250     OptionsInfo["psi4"] = Psi4Util.InitializePsi4(Psi4RunParams = OptionsInfo["Psi4RunParams"], Psi4OptionsParams = OptionsInfo["Psi4OptionsParams"], PrintVersion = False, PrintHeader = True)
 251     
 252     # Setup conversion factor for energy units...
 253     SetupEnergyConversionFactor(OptionsInfo["psi4"])
 254     
 255 def WorkerProcess(EncodedMolInfo):
 256     """Process data for a worker process."""
 257     
 258     if not OptionsInfo["Psi4Initialized"]:
 259         InitializePsi4ForWorkerProcess()
 260     
 261     MolIndex, EncodedMol = EncodedMolInfo
 262     
 263     CalcStatus = False
 264     Energy = None
 265     
 266     if EncodedMol is None:
 267         return [MolIndex, None, CalcStatus, Energy]
 268     
 269     Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
 270     MolCount = MolIndex + 1
 271     
 272     if not CheckAndValidateMolecule(Mol, MolCount):
 273         return [MolIndex, None, CalcStatus, Energy]
 274     
 275     # Setup a Psi4 molecule...
 276     Psi4Mol = SetupPsi4Mol(OptionsInfo["psi4"], Mol, MolCount)
 277     if Psi4Mol is None:
 278         return [MolIndex, None, CalcStatus, Energy]
 279     
 280     CalcStatus, Energy = CalculateMoleculeEnergy(OptionsInfo["psi4"], Psi4Mol, Mol, MolCount)
 281     
 282     return [MolIndex, RDKitUtil.MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.MolProps | Chem.PropertyPickleOptions.PrivateProps), CalcStatus, Energy]
 283 
 284 def CalculateMoleculeEnergy(Psi4Handle, Psi4Mol, Mol, MolNum = None):
 285     """Calculate energy."""
 286     
 287     Status = False
 288     Energy = None
 289     
 290     #  Setup reference wave function...
 291     Reference = SetupReferenceWavefunction(Mol)
 292     Psi4Handle.set_options({'Reference': Reference})
 293     
 294     # Setup method name and basis set...
 295     MethodName, BasisSet = SetupMethodNameAndBasisSet(Mol)
 296     
 297     Status, Energy = Psi4Util.CalculateSinglePointEnergy(Psi4Handle, Psi4Mol, MethodName, BasisSet, Quiet = OptionsInfo["QuietMode"])
 298     
 299     # Convert energy units...
 300     if Status:
 301         if OptionsInfo["ApplyEnergyConversionFactor"]:
 302             Energy = Energy * OptionsInfo["EnergyConversionFactor"]
 303 
 304     # Clean up
 305     PerformPsi4Cleanup(Psi4Handle)
 306     
 307     return (Status, Energy)
 308 
 309 def SetupPsi4Mol(Psi4Handle, Mol, MolCount = None):
 310     """Setup a Psi4 molecule object."""
 311     
 312     MolGeometry = RDKitUtil.GetPsi4XYZFormatString(Mol, NoCom = True, NoReorient = True)
 313     
 314     try:
 315         Psi4Mol = Psi4Handle.geometry(MolGeometry)
 316     except Exception as ErrMsg:
 317         Psi4Mol = None
 318         if not OptionsInfo["QuietMode"]:
 319             MiscUtil.PrintWarning("Failed to create Psi4 molecule from geometry string: %s\n" % ErrMsg)
 320             MolName = RDKitUtil.GetMolName(Mol, MolCount)
 321             MiscUtil.PrintWarning("Ignoring molecule: %s" % MolName)
 322     
 323     return Psi4Mol
 324 
 325 def PerformPsi4Cleanup(Psi4Handle):
 326     """Perform clean up."""
 327     
 328     # Clean up after Psi4 run ...
 329     Psi4Handle.core.clean()
 330     
 331     # Clean up any leftover scratch files...
 332     if OptionsInfo["MPMode"]:
 333         Psi4Util.RemoveScratchFiles(Psi4Handle, OptionsInfo["Psi4RunParams"]["OutputFile"])

 335 def CheckAndValidateMolecule(Mol, MolCount = None):
 336     """Validate molecule for Psi4 calculations."""
 337     
 338     if Mol is None:
 339         if not OptionsInfo["QuietMode"]:
 340             MiscUtil.PrintInfo("\nProcessing molecule number %s..." % MolCount)
 341         return False
 342     
 343     MolName = RDKitUtil.GetMolName(Mol, MolCount)
 344     if not OptionsInfo["QuietMode"]:
 345         MiscUtil.PrintInfo("\nProcessing molecule %s..." % MolName)
 346     
 347     if RDKitUtil.IsMolEmpty(Mol):
 348         if not OptionsInfo["QuietMode"]:
 349             MiscUtil.PrintWarning("Ignoring empty molecule: %s\n" % MolName)
 350         return False
 351     
 352     if not RDKitUtil.ValidateElementSymbols(RDKitUtil.GetAtomSymbols(Mol)):
 353         if not OptionsInfo["QuietMode"]:
 354             MiscUtil.PrintWarning("Ignoring molecule containing invalid element symbols: %s\n" % MolName)
 355         return False
 356     
 357     # Check for 3D flag...
 358     if not Mol.GetConformer().Is3D():
 359         if not OptionsInfo["QuietMode"]:
 360             MiscUtil.PrintWarning("3D tag is not set for molecule: %s\n" % MolName)
 361     
 362     # Check for missing hydrogens...
 363     if RDKitUtil.AreHydrogensMissingInMolecule(Mol):
 364         if not OptionsInfo["QuietMode"]:
 365             MiscUtil.PrintWarning("Missing hydrogens in molecule: %s\n" % MolName)
 366     
 367     return True
 368 
 369 def SetupMethodNameAndBasisSet(Mol):
 370     """Setup method name and basis set."""
 371     
 372     MethodName = OptionsInfo["MethodName"]
 373     if OptionsInfo["MethodNameAuto"]:
 374         MethodName = "B3LYP"
 375     
 376     BasisSet = OptionsInfo["BasisSet"]
 377     if OptionsInfo["BasisSetAuto"]:
 378         BasisSet = "6-31+G**" if RDKitUtil.IsAtomSymbolPresentInMol(Mol, "S") else "6-31G**"
 379     
 380     return (MethodName, BasisSet)
 381 
 382 def SetupReferenceWavefunction(Mol):
 383     """Setup reference wavefunction."""
 384     
 385     Reference = OptionsInfo["Reference"]
 386     if OptionsInfo["ReferenceAuto"]:
 387         Reference = 'UHF' if (RDKitUtil.GetSpinMultiplicity(Mol) > 1) else 'RHF'
 388     
 389     return Reference

 391 def SetupEnergyConversionFactor(Psi4Handle):
 392     """Setup converstion factor for energt units. The Psi4 energy units are Hartrees."""
 393     
 394     EnergyUnits = OptionsInfo["EnergyUnits"]
 395     
 396     ApplyConversionFactor = True
 397     if re.match("^kcal\/mol$", EnergyUnits, re.I):
 398         ConversionFactor = Psi4Handle.constants.hartree2kcalmol
 399     elif re.match("^kJ\/mol$", EnergyUnits, re.I):
 400         ConversionFactor = Psi4Handle.constants.hartree2kJmol
 401     elif re.match("^eV$", EnergyUnits, re.I):
 402         ConversionFactor = Psi4Handle.constants.hartree2ev
 403     else:
 404         ApplyConversionFactor = False
 405         ConversionFactor = 1.0
 406     
 407     OptionsInfo["ApplyEnergyConversionFactor"] = ApplyConversionFactor
 408     OptionsInfo["EnergyConversionFactor"] = ConversionFactor
 409 
 410 def WriteMolecule(Writer, Mol, Energy):
 411     """Write molecule."""
 412     
 413     Mol.SetProp(OptionsInfo["EnergyDataFieldLabel"], Energy)
 414     Writer.write(Mol)

 416 def ProcessOptions():
 417     """Process and validate command line arguments and options."""
 418     
 419     MiscUtil.PrintInfo("Processing options...")
 420     
 421     # Validate options...
 422     ValidateOptions()
 423     
 424     OptionsInfo["Infile"] = Options["--infile"]
 425     ParamsDefaultInfoOverride = {"RemoveHydrogens": False}
 426     OptionsInfo["InfileParams"] = MiscUtil.ProcessOptionInfileParameters("--infileParams", Options["--infileParams"], InfileName = Options["--infile"], ParamsDefaultInfo = ParamsDefaultInfoOverride)
 427     
 428     OptionsInfo["Outfile"] = Options["--outfile"]
 429     OptionsInfo["OutfileParams"] = MiscUtil.ProcessOptionOutfileParameters("--outfileParams", Options["--outfileParams"])
 430     
 431     OptionsInfo["Overwrite"] = Options["--overwrite"]
 432     
 433     # Method, basis set, and reference wavefunction...
 434     OptionsInfo["BasisSet"] = Options["--basisSet"]
 435     OptionsInfo["BasisSetAuto"] = True if re.match("^auto$", Options["--basisSet"], re.I) else False
 436     
 437     OptionsInfo["MethodName"] = Options["--methodName"]
 438     OptionsInfo["MethodNameAuto"] = True if re.match("^auto$", Options["--methodName"], re.I) else False
 439     
 440     OptionsInfo["Reference"] = Options["--reference"]
 441     OptionsInfo["ReferenceAuto"] = True if re.match("^auto$", Options["--reference"], re.I) else False
 442     
 443     # Run and options parameters...
 444     OptionsInfo["Psi4OptionsParams"] = Psi4Util.ProcessPsi4OptionsParameters("--psi4OptionsParams", Options["--psi4OptionsParams"])
 445     OptionsInfo["Psi4RunParams"] = Psi4Util.ProcessPsi4RunParameters("--psi4RunParams", Options["--psi4RunParams"], InfileName = OptionsInfo["Infile"])
 446 
 447     # Energy units and label...
 448     OptionsInfo["EnergyUnits"] = Options["--energyUnits"]
 449     EnergyDataFieldLabel = Options["--energyDataFieldLabel"]
 450     if re.match("^auto$", EnergyDataFieldLabel, re.I):
 451         EnergyDataFieldLabel = "Psi4_Energy (%s)" % Options["--energyUnits"]
 452     OptionsInfo["EnergyDataFieldLabel"] = EnergyDataFieldLabel
 453 
 454     OptionsInfo["MPMode"] = True if re.match("^yes$", Options["--mp"], re.I) else False
 455     OptionsInfo["MPParams"] = MiscUtil.ProcessOptionMultiprocessingParameters("--mpParams", Options["--mpParams"])
 456     
 457     OptionsInfo["Precision"] = int(Options["--precision"])
 458     OptionsInfo["QuietMode"] = True if re.match("^yes$", Options["--quiet"], re.I) else False

 566             M06, M06-2x, M06-HF, M06-L, MN12-L, MN15, MN15-D3BJ,PBE, PBE0,
 567             PBEH3c, PW6B95, PW6B95-D3BJ, WB97, WB97X, WB97X-D, WB97X-D3BJ
 568             
 569     --mp <yes or no>  [default: no]
 570         Use multiprocessing.
 571          
 572         By default, input data is retrieved in a lazy manner via mp.Pool.imap()
 573         function employing lazy RDKit data iterable. This allows processing of
 574         arbitrary large data sets without any additional requirements memory.
 575         
 576         All input data may be optionally loaded into memory by mp.Pool.map()
 577         before starting worker processes in a process pool by setting the value
 578         of 'inputDataMode' to 'InMemory' in '--mpParams' option.
 579         
 580         A word to the wise: The default 'chunkSize' value of 1 during 'Lazy' input
 581         data mode may adversely impact the performance. The '--mpParams' section
 582         provides additional information to tune the value of 'chunkSize'.
 583     --mpParams <Name,Value,...>  [default: auto]
 584         A comma delimited list of parameter name and value pairs to configure
 585         multiprocessing.
 586         
 587         The supported parameter names along with their default and possible
 588         values are shown below:
 589         
 590             chunkSize, auto
 591             inputDataMode, Lazy   [ Possible values: InMemory or Lazy ]
 592             numProcesses, auto   [ Default: mp.cpu_count() ]
 593         
 594         These parameters are used by the following functions to configure and
 595         control the behavior of multiprocessing: mp.Pool(), mp.Pool.map(), and
 596         mp.Pool.imap().
 597         
 598         The chunkSize determines chunks of input data passed to each worker
 599         process in a process pool by mp.Pool.map() and mp.Pool.imap() functions.
 600         The default value of chunkSize is dependent on the value of 'inputDataMode'.
 601         
 602         The mp.Pool.map() function, invoked during 'InMemory' input data mode,
 603         automatically converts RDKit data iterable into a list, loads all data into
 604         memory, and calculates the default chunkSize using the following method
 605         as shown in its code:
 606         
 607             chunkSize, extra = divmod(len(dataIterable), len(numProcesses) * 4)
 608             if extra: chunkSize += 1
 609         
 610         For example, the default chunkSize will be 7 for a pool of 4 worker processes
 611         and 100 data items.
 612         
 613         The mp.Pool.imap() function, invoked during 'Lazy' input data mode, employs
 614         'lazy' RDKit data iterable to retrieve data as needed, without loading all the
 615         data into memory. Consequently, the size of input data is not known a priori.
 616         It's not possible to estimate an optimal value for the chunkSize. The default 
 617         chunkSize is set to 1.
 618         
 619         The default value for the chunkSize during 'Lazy' data mode may adversely
 620         impact the performance due to the overhead associated with exchanging
 621         small chunks of data. It is generally a good idea to explicitly set chunkSize to
 622         a larger value during 'Lazy' input data mode, based on the size of your input
 623         data and number of processes in the process pool.
 624         
 625         The mp.Pool.map() function waits for all worker processes to process all
 626         the data and return the results. The mp.Pool.imap() function, however,
 627         returns the the results obtained from worker processes as soon as the
 628         results become available for specified chunks of data.
 629         
 630         The order of data in the results returned by both mp.Pool.map() and 
 631         mp.Pool.imap() functions always corresponds to the input data.
 632     -o, --outfile <outfile>
 633         Output file name.
 634     --outfileParams <Name,Value,...>  [default: auto]
 635         A comma delimited list of parameter name and value pairs for writing
 636         molecules to files. The supported parameter names for different file
 637         formats, along with their default values, are shown below:
 638             
 639             SD: kekulize,yes
 640             
 641     --overwrite
 642         Overwrite existing files.
 643     --precision <number>  [default: 6]

 645     --psi4OptionsParams <Name,Value,...>  [default: none]
 646         A comma delimited list of Psi4 option name and value pairs for setting
 647         global and module options. The names are 'option_name' for global options
 648         and 'module_name__option_name' for options local to a module. The
 649         specified option names must be valid Psi4 names. No validation is
 650         performed.
 651         
 652         The specified option name and  value pairs are processed and passed to
 653         psi4.set_options() as a dictionary. The supported value types are float,
 654         integer, boolean, or string. The float value string is converted into a float.
 655         The valid values for a boolean string are yes, no, true, false, on, or off. 
 656     --psi4RunParams <Name,Value,...>  [default: auto]
 657         A comma delimited list of parameter name and value pairs for configuring
 658         Psi4 jobs.
 659         
 660         The supported parameter names along with their default and possible
 661         values are shown below:
 662              
 663             MemoryInGB, 1
 664             NumThreads, 1
 665             OutputFile, auto   [ Possible  values: stdout, quiet, or FileName ]
 666             ScratchDir, auto   [ Possivle values: DirName]
 667             RemoveOutputFile, yes   [ Possible values: yes, no, true, or false]
 668             
 669         These parameters control the runtime behavior of Psi4.
 670         
 671         The default file name for 'OutputFile' is <InFileRoot>_Psi4.out. The PID
 672         is appended to output file name during multiprocessing as shown:
 673         <InFileRoot>_Psi4_<PIDNum>.out. The 'stdout' value for 'OutputType'
 674         sends Psi4 output to stdout. The 'quiet' or 'devnull' value suppresses
 675         all Psi4 output.
 676         
 677         The default 'Yes' value of 'RemoveOutputFile' option forces the removal
 678         of any existing Psi4 before creating new files to append output from
 679         multiple Psi4 runs.
 680         
 681         The option 'ScratchDir' is a directory path to the location of scratch
 682         files. The default value corresponds to Psi4 default. It may be used to
 683         override the deafult path.
 684     -q, --quiet <yes or no>  [default: no]
 685         Use quiet mode. The warning and information messages will not be printed.
 686     -r, --reference <text>  [default: auto]
 687         Reference wave function to use for energy calculation. Default: RHF or UHF.
 688         The default values are Restricted Hartree-Fock (RHF) for closed-shell molecules
 689         with all electrons paired and Unrestricted Hartree-Fock (UHF) for open-shell
 690         molecules with unpaired electrons.
 691         
 692         The specified value must be a valid Psi4 reference wave function. No validation
 693         is performed. For example: ROHF, CUHF, RKS, etc.
 694         
 695         The spin multiplicity determines the default value of reference wave function
 696         for input molecules. It is calculated from number of free radical electrons using
 697         Hund's rule of maximum multiplicity defined as 2S + 1 where S is the total
 698         electron spin. The total spin is 1/2 the number of free radical electrons in a 
 699         molecule. The value of 'SpinMultiplicity' molecule property takes precedence
 700         over the calculated value of spin multiplicity.
