  82 def StandardizeMolecules():
  83     """Stanardize molecules."""
  84     
  85     # Setup a molecule reader...
  86     MiscUtil.PrintInfo("\nProcessing file %s..." % OptionsInfo["Infile"])
  87     Mols  = RDKitUtil.ReadMolecules(OptionsInfo["Infile"], **OptionsInfo["InfileParams"])
  88     
  89     # Set up a molecule writer...
  90     Writer = SetupMoleculeWriter()
  91 
  92     MolCount, ValidMolCount, StandardizationFailedCount = ProcessMolecules(Mols, Writer)
  93 
  94     if Writer is not None:
  95         Writer.close()
  96     
  97     MiscUtil.PrintInfo("\nTotal number of molecules: %d" % MolCount)
  98     MiscUtil.PrintInfo("Number of valid molecules: %d" % ValidMolCount)
  99     MiscUtil.PrintInfo("Number of molecules failed during standardization: %d" % StandardizationFailedCount)
 100     MiscUtil.PrintInfo("Number of ignored molecules: %d" % (MolCount - ValidMolCount + StandardizationFailedCount))
 101     
 102     MiscUtil.PrintInfo("\nNumber of standardized molecules: %d" % (ValidMolCount - StandardizationFailedCount))

 104 def ProcessMolecules(Mols, Writer):
 105     """Process and standardize molecules."""
 106 
 107     if OptionsInfo["MPMode"]:
 108         return ProcessMoleculesUsingMultipleProcesses(Mols, Writer)
 109     else:
 110         return ProcessMoleculesUsingSingleProcess(Mols, Writer)
 111 
 112 def ProcessMoleculesUsingSingleProcess(Mols,  Writer):
 113     """Process and standardize molecules using a single process."""
 114 
 115     MiscUtil.PrintInfo("\nStandardizing molecules...")
 116     
 117     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 118     SetSMILESMolProps = OptionsInfo["OutfileParams"]["SetSMILESMolProps"]
 119 
 120     # Set up standardize...
 121     SetupStandardize()
 122 
 123     (MolCount, ValidMolCount, StandardizationFailedCount) = [0] * 3
 124     FirstMol = True
 125     for Mol in Mols:
 126         MolCount += 1
 127         
 128         if Mol is None:
 129             continue
 130         
 131         if RDKitUtil.IsMolEmpty(Mol):
 132             if not OptionsInfo["QuietMode"]:
 133                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
 134                 MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 135             continue
 136         
 137         ValidMolCount += 1
 138         if FirstMol:
 139             FirstMol = False
 140             if SetSMILESMolProps:
 141                 RDKitUtil.SetWriterMolProps(Writer, Mol)
 142         
 143         StandardizedMol,  StandardizationStatus = PerformStandardization(Mol, MolCount)
 144         if not StandardizationStatus:
 145             if not OptionsInfo["QuietMode"]:
 146                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
 147                 MiscUtil.PrintWarning("Failed to standardize molecule %s" % MolName)
 148             
 149             StandardizationFailedCount += 1
 150             continue
 151         
 152         WriteMolecule(Writer, StandardizedMol, Compute2DCoords)
 153     
 154     return (MolCount, ValidMolCount, StandardizationFailedCount)
 155     
 156 def ProcessMoleculesUsingMultipleProcesses(Mols, Writer):
 157     """Process and standardize molecules using  multiprocessing."""
 158     
 159     MiscUtil.PrintInfo("\nStandardize molecules using multiprocessing...")
 160     
 161     MPParams = OptionsInfo["MPParams"]
 162     Compute2DCoords = OptionsInfo["OutfileParams"]["Compute2DCoords"]
 163     
 164     # Setup data for initializing a worker process...
 165     InitializeWorkerProcessArgs = (MiscUtil.ObjectToBase64EncodedString(Options), MiscUtil.ObjectToBase64EncodedString(OptionsInfo))
 166 
 167     # Setup a encoded mols data iterable for a worker process by pickling only public
 168     # and private molecule properties...
 169     WorkerProcessDataIterable = RDKitUtil.GenerateBase64EncodedMolStrings(Mols)
 170 
 171     # Setup process pool along with data initialization for each process...
 172     MiscUtil.PrintInfo("\nConfiguring multiprocessing using %s method..." % ("mp.Pool.imap()" if re.match("^Lazy$", MPParams["InputDataMode"], re.I) else "mp.Pool.map()"))
 173     MiscUtil.PrintInfo("NumProcesses: %s; InputDataMode: %s; ChunkSize: %s\n" % (MPParams["NumProcesses"], MPParams["InputDataMode"], ("automatic" if MPParams["ChunkSize"] is None else MPParams["ChunkSize"])))
 174     
 175     ProcessPool = mp.Pool(MPParams["NumProcesses"], InitializeWorkerProcess, InitializeWorkerProcessArgs)
 176     
 177     # Start processing...
 178     if re.match("^Lazy$", MPParams["InputDataMode"], re.I):
 179         Results = ProcessPool.imap(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
 180     elif re.match("^InMemory$", MPParams["InputDataMode"], re.I):
 181         Results = ProcessPool.map(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
 182     else:
 183         MiscUtil.PrintError("The value, %s, specified for \"--inputDataMode\" is not supported." % (MPParams["InputDataMode"]))
 184     
 185     SetSMILESMolProps = OptionsInfo["OutfileParams"]["SetSMILESMolProps"]
 186     
 187     (MolCount, ValidMolCount, StandardizationFailedCount) = [0] * 3
 188     FirstMol = True
 189     for Result in Results:
 190         MolCount += 1
 191         MolIndex, EncodedMol, StandardizationStatus = Result
 192         
 193         if EncodedMol is None:
 194             continue
 195         ValidMolCount += 1
 196         
 197         Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
 198         
 199         if FirstMol:
 200             FirstMol = False
 201             if SetSMILESMolProps:
 202                 RDKitUtil.SetWriterMolProps(Writer, Mol)
 203         
 204         if not StandardizationStatus:
 205             if not OptionsInfo["QuietMode"]:
 206                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
 207                 MiscUtil.PrintWarning("Failed to standardize molecule %s" % MolName)
 208             
 209             StandardizationFailedCount += 1
 210             continue
 211         
 212         WriteMolecule(Writer, Mol, Compute2DCoords)
 213     
 214     return (MolCount, ValidMolCount, StandardizationFailedCount)

 216 def InitializeWorkerProcess(*EncodedArgs):
 217     """Initialize data for a worker process."""
 218 
 219     global Options, OptionsInfo
 220     
 221     MiscUtil.PrintInfo("Starting process (PID: %s)..." % os.getpid())
 222 
 223     # Decode Options and OptionInfo...
 224     Options = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[0])
 225     OptionsInfo = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[1])
 226 
 227     # Set up standardize...
 228     SetupStandardize()
 229 
 230 def WorkerProcess(EncodedMolInfo):
 231     """Process data for a worker process."""
 232     
 233     MolIndex, EncodedMol = EncodedMolInfo
 234     
 235     if EncodedMol is None:
 236         return [MolIndex, None, False]
 237         
 238     Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
 239     if RDKitUtil.IsMolEmpty(Mol):
 240         if not OptionsInfo["QuietMode"]:
 241             MolName = RDKitUtil.GetMolName(Mol, (MolIndex + 1))
 242             MiscUtil.PrintWarning("Ignoring empty molecule: %s" % MolName)
 243         return [MolIndex, None, False]
 244     
 245     Mol, StandardizationStatus = PerformStandardization(Mol, (MolIndex + 1))
 246     EncodedMol = RDKitUtil.MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.MolProps | Chem.PropertyPickleOptions.PrivateProps)
 247     
 248     return [MolIndex, EncodedMol, StandardizationStatus]

 250 def PerformStandardization(Mol, MolNum):
 251     """Perform standardization and return a standardized mol along with the status of
 252     the standardization."""
 253 
 254     try:
 255         # Step 1: Cleanup...
 256         if OptionsInfo["MethodologyParams"]["Cleanup"]:
 257             Mol = CleanupMolecule(Mol)
 258         
 259         # Step2: Get largest fragment...
 260         if OptionsInfo["MethodologyParams"]["RemoveFragments"]:
 261             Mol = ChooseLargestMoleculeFragment(Mol)
 262         
 263         # Step3: Neutralize...
 264         if OptionsInfo["MethodologyParams"]["Neutralize"]:
 265             Mol = NeutralizeMolecule(Mol)
 266         
 267         # Step4: Canonicalize tautomer...
 268         if OptionsInfo["MethodologyParams"]["CanonicalizeTautomer"]:
 269             Mol = CanonicalizeMoleculeTautomer(Mol)
 270         
 271         Status = True
 272     except Exception as ErrMsg:
 273         if not OptionsInfo["QuietMode"]:
 274             MiscUtil.PrintWarning("Failed to standardize molecule %s: %s" % (RDKitUtil.GetMolName(Mol, MolNum), ErrMsg))
 275         Status = False
 276 
 277     return (Mol, Status)

 279 def CleanupMolecule(Mol):
 280     """Clean up molecule."""
 281     
 282     if OptionsInfo["StandardizeParams"]["CleanupRemoveHydrogens"]:
 283         Mol = Chem.RemoveHs(Mol)
 284         
 285     if OptionsInfo["StandardizeParams"]["CleanupDisconnectMetals"]:
 286         # Disconnect metal atoms that are defined as covalently bonded to non-metals...
 287         Mol = OptionsInfo["StandardizeObjects"]["MetalDisconnector"].Disconnect(Mol)
 288     
 289     if OptionsInfo["StandardizeParams"]["CleanupNormalize"]:
 290         # Apply normalization transforms to correct functional groups and recombine charges...
 291         Mol = rdMolStandardize.Normalize(Mol, OptionsInfo["CleanupParams"])
 292     
 293     if OptionsInfo["StandardizeParams"]["CleanupReionize"]:
 294         # Ensure the strongest acid groups ionize first in partially ionized molecules...
 295         Mol = rdMolStandardize.Reionize(Mol, OptionsInfo["CleanupParams"])
 296     
 297     if OptionsInfo["StandardizeParams"]["CleanupAssignStereo"]:
 298         # Assign stereochemistry
 299         Chem.AssignStereochemistry(Mol, force=OptionsInfo["StandardizeParams"]["CleanupAssignStereoForce"], cleanIt=OptionsInfo["StandardizeParams"]["CleanupAssignStereoCleanIt"])
 300     
 301     Mol.UpdatePropertyCache()
 302 
 303     return Mol

 305 def ChooseLargestMoleculeFragment(Mol):
 306     """Choose largest molecule fragment. """
 307 
 308     return OptionsInfo["StandardizeObjects"]["LargestFragmentChooser"].choose(Mol)
 309 
 310 def NeutralizeMolecule(Mol):
 311     """Neutralize molecule."""
 312     
 313     return OptionsInfo["StandardizeObjects"]["Uncharger"].uncharge(Mol)
 314 
 315 def CanonicalizeMoleculeTautomer(Mol):
 316     """Canonicalize molecule tautomer."""
 317     
 318     return OptionsInfo["StandardizeObjects"]["TautomerEnumerator"].Canonicalize(Mol)

 320 def SetupStandardize():
 321     """Setup RDKit standardize objects to perform standardization."""
 322 
 323     OptionsInfo["StandardizeObjects"] = {}
 324     
 325     OptionsInfo["CleanupParams"] = SetupStandardizeCleanupParameters()
 326     
 327     if OptionsInfo["MethodologyParams"]["Cleanup"]:
 328         if OptionsInfo["StandardizeParams"]["CleanupDisconnectMetals"]:
 329             OptionsInfo["StandardizeObjects"]["MetalDisconnector"] = rdMolStandardize.MetalDisconnector()
 330     
 331     if OptionsInfo["MethodologyParams"]["RemoveFragments"]:
 332         OptionsInfo["StandardizeObjects"]["LargestFragmentChooser"] = rdMolStandardize.LargestFragmentChooser(OptionsInfo["CleanupParams"])
 333     
 334     if OptionsInfo["MethodologyParams"]["Neutralize"]:
 335         OptionsInfo["StandardizeObjects"]["Uncharger"] = rdMolStandardize.Uncharger(OptionsInfo["CleanupParams"].doCanonical)
 336 
 337     if OptionsInfo["MethodologyParams"]["CanonicalizeTautomer"]:
 338         OptionsInfo["StandardizeObjects"]["TautomerEnumerator"] = rdMolStandardize.TautomerEnumerator(OptionsInfo["CleanupParams"])
 339 
 340 def SetupStandardizeCleanupParameters():
 341     """Setup standardize clean up parameters for RDKit. """
 342 
 343     CleanupParams = rdMolStandardize.CleanupParameters()
 344     StandardizeParams = OptionsInfo["StandardizeParams"]
 345     
 346     if StandardizeParams["AcidBaseFile"] is not None:
 347         CleanupParams.acidbaseFile = StandardizeParams["AcidBaseFile"]
 348     if StandardizeParams["FragmentFile"] is not None:
 349         CleanupParams.acidbaseFile = StandardizeParams["FragmentFile"]
 350     if StandardizeParams["NormalizationsFile"] is not None:
 351         CleanupParams.normalizationsFile = StandardizeParams["NormalizationsFile"]
 352     if StandardizeParams["TautomerTransformsFile"] is not None:
 353         CleanupParams.tautomerTransformsFile = StandardizeParams["TautomerTransformsFile"]
 354     
 355     CleanupParams.maxRestarts = StandardizeParams["CleanupNormalizeMaxRestarts"]
 356     
 357     CleanupParams.doCanonical = StandardizeParams["DoCanonical"]
 358     
 359     CleanupParams.largestFragmentChooserUseAtomCount = StandardizeParams["LargestFragmentChooserUseAtomCount"]
 360     CleanupParams.largestFragmentChooserCountHeavyAtomsOnly = StandardizeParams["LargestFragmentChooserCountHeavyAtomsOnly"]
 361     
 362     CleanupParams.preferOrganic = StandardizeParams["PreferOrganic"]
 363     
 364     CleanupParams.maxTautomers = StandardizeParams["MaxTautomers"]
 365     CleanupParams.maxTransforms = StandardizeParams["MaxTransforms"]
 366     CleanupParams.tautomerRemoveBondStereo = StandardizeParams["TautomerRemoveBondStereo"]
 367     CleanupParams.tautomerRemoveIsotopicHs = StandardizeParams["TautomerRemoveIsotopicHs"]
 368     CleanupParams.tautomerRemoveSp3Stereo = StandardizeParams["TautomerRemoveSp3Stereo"]
 369     CleanupParams.tautomerReassignStereo = StandardizeParams["TautomerReassignStereo"]
 370     
 371     return CleanupParams

 373 def WriteMolecule(Writer, Mol, Compute2DCoords):
 374     """Write out molecule."""
 375     
 376     if OptionsInfo["CountMode"]:
 377         return
 378     
 379     if Compute2DCoords:
 380         AllChem.Compute2DCoords(Mol)
 381     
 382     Writer.write(Mol)

 384 def SetupMoleculeWriter():
 385     """Setup a molecule writer."""
 386     
 387     Writer = None
 388     if OptionsInfo["CountMode"]:
 389         return Writer
 390 
 391     Writer = RDKitUtil.MoleculesWriter(OptionsInfo["Outfile"], **OptionsInfo["OutfileParams"])
 392     if Writer is None:
 393         MiscUtil.PrintError("Failed to setup a writer for output fie %s " % OptionsInfo["Outfile"])
 394     MiscUtil.PrintInfo("Generating file %s..." % OptionsInfo["Outfile"])
 395     
 396     return Writer

 570         The step2, removeFragments, employs rdMolStandardize.FragmentParent()
 571         function to keep the largest fragment.
 572         
 573         The step3, neutralize, uses rdMolStandardize.Uncharger().uncharge()
 574         function to neutralize molecules by adding/removing hydrogens.
 575         
 576         The step4, canonicalizeTautomer, relies on Canonicalize() function availabe via
 577         rdMolStandardize.TautomerEnumerator() to select a canonical tautomer.
 578     --mp <yes or no>  [default: no]
 579         Use multiprocessing.
 580          
 581         By default, input data is retrieved in a lazy manner via mp.Pool.imap()
 582         function employing lazy RDKit data iterable. This allows processing of
 583         arbitrary large data sets without any additional requirements memory.
 584         
 585         All input data may be optionally loaded into memory by mp.Pool.map()
 586         before starting worker processes in a process pool by setting the value
 587         of 'inputDataMode' to 'InMemory' in '--mpParams' option.
 588         
 589         A word to the wise: The default 'chunkSize' value of 1 during 'Lazy' input
 590         data mode may adversely impact the performance. The '--mpParams' section
 591         provides additional information to tune the value of 'chunkSize'.
 592     --mpParams <Name,Value,...>  [default: auto]
 593         A comma delimited list of parameter name and value pairs to configure
 594         multiprocessing.
 595         
 596         The supported parameter names along with their default and possible
 597         values are shown below:
 598         
 599             chunkSize, auto
 600             inputDataMode, Lazy   [ Possible values: InMemory or Lazy ]
 601             numProcesses, auto   [ Default: mp.cpu_count() ]
 602         
 603         These parameters are used by the following functions to configure and
 604         control the behavior of multiprocessing: mp.Pool(), mp.Pool.map(), and
 605         mp.Pool.imap().
 606         
 607         The chunkSize determines chunks of input data passed to each worker
 608         process in a process pool by mp.Pool.map() and mp.Pool.imap() functions.
 609         The default value of chunkSize is dependent on the value of 'inputDataMode'.
 610         
 611         The mp.Pool.map() function, invoked during 'InMemory' input data mode,
 612         automatically converts RDKit data iterable into a list, loads all data into
 613         memory, and calculates the default chunkSize using the following method
 614         as shown in its code:
 615         
 616             chunkSize, extra = divmod(len(dataIterable), len(numProcesses) * 4)
 617             if extra: chunkSize += 1
 618         
 619         For example, the default chunkSize will be 7 for a pool of 4 worker processes
 620         and 100 data items.
 621         
 622         The mp.Pool.imap() function, invoked during 'Lazy' input data mode, employs
 623         'lazy' RDKit data iterable to retrieve data as needed, without loading all the
 624         data into memory. Consequently, the size of input data is not known a priori.
 625         It's not possible to estimate an optimal value for the chunkSize. The default 
 626         chunkSize is set to 1.
 627         
 628         The default value for the chunkSize during 'Lazy' data mode may adversely
 629         impact the performance due to the overhead associated with exchanging
 630         small chunks of data. It is generally a good idea to explicitly set chunkSize to
 631         a larger value during 'Lazy' input data mode, based on the size of your input
 632         data and number of processes in the process pool.
 633         
 634         The mp.Pool.map() function waits for all worker processes to process all
 635         the data and return the results. The mp.Pool.imap() function, however,
 636         returns the the results obtained from worker processes as soon as the
 637         results become available for specified chunks of data.
 638         
 639         The order of data in the results returned by both mp.Pool.map() and 
 640         mp.Pool.imap() functions always corresponds to the input data.
 641     -o, --outfile <outfile>
 642         Output file name.
 643     --outfileParams <Name,Value,...>  [default: auto]
 644         A comma delimited list of parameter name and value pairs for writing
 645         molecules to files. The supported parameter names for different file
 646         formats, along with their default values, are shown below:
 647             
 648             SD: compute2DCoords,auto,kekulize,yes
 649             SMILES: smilesKekulize,no,smilesDelimiter,space, smilesIsomeric,yes,
 650                 smilesTitleLine,yes,smilesMolName,yes,smilesMolProps,no
 651             
 652         Default value for compute2DCoords: yes for SMILES input file; no for all other
 653         file types.

 655         Overwrite existing files.

 657         Use quiet mode. The warning and information messages will not be printed.
 658     -s, --standardizeParams <Name,Value,...>  [default: auto]
 659         A comma delimited list of parameter name and value pairs for standardizing
 660         molecules. The supported parameter names along with their default values
 661         are shown below:
 662             
 663             acidbaseFile,none,fragmentFile,none,normalizationsFile,none,
 664             tautomerTransformsFile,none,
 665             cleanupRemoveHydrogens,yes,cleanupDisconnectMetals,yes,
 666             cleanupNormalize,yes,cleanupNormalizeMaxRestarts,200,
 667             cleanupReionize,yes,cleanupAssignStereo,yes,
 668             cleanupAssignStereoCleanIt,yes,cleanupAssignStereoForce,yes
 669             largestFragmentChooserUseAtomCount,yes,
 670             largestFragmentChooserCountHeavyAtomsOnly,no,preferOrganic,no,
 671             doCanonical,yes,
 672             maxTautomers,1000,maxTransforms,1000,
 673             tautomerRemoveBondStereo,yes,tautomerRemoveIsotopicHs,yes
 674             tautomerRemoveSp3Stereo,yes,tautomerReassignStereo,yes
 675             
 676         A brief description of the standardization parameters, taken from RDKit
 677         documentation, is as follows:
 678             
 679             acidbaseFile - File containing acid and base definitions
 680             fragmentFile - File containing fragment definitions
 681             normalizationsFile - File conataining normalization transformations
 682             tautomerTransformsFile - File containing tautomer transformations
 683             
 684             cleanupRemoveHydrogens - Remove hydrogens druring cleanup
 685             cleanupDisconnectMetals - Disconnect metal atoms covalently bonded
 686                 to non-metals during cleanup
 687             cleanupNormalize - Normalize functional groups and recombine
 688                 charges during cleanup
 689             cleanupNormalizeMaxRestarts - Maximum number of restarts during
 690                 normalization step of cleanup
 691             cleanupReionize -Ionize strongest acid groups first in partially
 692                 ionized molecules during cleanup
 693             cleanupAssignStereo - Assign stererochemistry during cleanup
 694             cleanupAssignStereoCleanIt - Clean property _CIPCode during
 695                 assign stereochemistry 
 696             cleanupAssignStereoForce - Always perform stereochemistry
 697                 calculation during assign stereochemistry
 698             
 699             largestFragmentChooserUseAtomCount - Use atom count as main
 700                 criterion before molecular weight to determine largest fragment
 701                 in LargestFragmentChooser
 702             largestFragmentChooserCountHeavyAtomsOnly - Count only heavy
 703                 atoms to determine largest fragment in LargestFragmentChooser
 704             preferOrganic - Prefer organic fragments over  inorganic ones when
 705                 choosing fragments
 706             
 707             doCanonical - Apply atom-order dependent normalizations in a
 708                 canonical order during uncharging
 709             
 710             maxTautomers - Maximum number of tautomers to generate
 711             maxTransforms - Maximum number of transforms to apply during
 712                 tautomer enumeration
 713             tautomerRemoveBondStereo - Remove stereochemistry from double bonds
 714                 involved in tautomerism
 715             tautomerRemoveIsotopicHs: Remove isotopic Hs from centers involved in tautomerism
 716             tautomerRemoveSp3Stereo - Remove stereochemistry from sp3 centers
 717                 involved in tautomerism
 718             tautomerReassignStereo - AssignStereochemistry on all generated tautomers
 719             
 720         The default value is set to none for the following  file name parameters:
 721         acidbaseFile, fragmentFile, normalizationsFile, and tautomerTransformsFile.
 722         The script relies on RDKit to automatically load appropriate acid base and
 723         fragment definitions along with normalization and tautomer transformations
 724         from a set of internal catalogs.
 725         
 726         Note: The fragmentFile doesn't appear to be used by the RDKit method
 727         rdMolStandardize.FragmentParent() to find largest fragment.
 728             
 729         The contents  of various standardization definitions and transformations files
 730         are described below:
 731

 733

 735                 -OSO3H      OS(=O)(=O)[OH]       OS(=O)(=O)[O-]
 736                 -SO3H       [!O]S(=O)(=O)[OH]    [!O]S(=O)(=O)[O-]

 743                 hydrogen     [H]

 746                 ... ... ...
 747         
 748             normalizationsFile - File conataining normalization transformations
 749             
 750                 // Name     SMIRKS
 751                 Sulfone to S(=O)(=O)        [S+2:1]([O-:2])([O-:3])>>
 752                     [S+0:1](=[O-0:2])(=[O-0:3])
 753                 Pyridine oxide to n+O-     [n:1]=[O:2]>>[n+:1][O-:2]
 754                 ... ... ...
 755         
 756             tautomerTransformsFile - File containing tautomer transformations
 757             
 758                 // Name                SMARTS   Bonds  Charges
 759                 1,3 (thio)keto/enol f  [CX4!H0]-[C]=[O,S,Se,Te;X1]

 762                 ... ... ...
 763
