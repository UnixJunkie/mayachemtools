  124 def ProcessMoleculesUsingSingleProcess(Mols, Writer):
  125     """Process and minimize molecules using a single process."""
  126     
  127     # Intialize Psi4...
  128     MiscUtil.PrintInfo("\nInitializing Psi4...")
  129     Psi4Handle = Psi4Util.InitializePsi4(Psi4RunParams = OptionsInfo["Psi4RunParams"], Psi4OptionsParams = OptionsInfo["Psi4OptionsParams"], PrintVersion = True, PrintHeader = True)
  130     OptionsInfo["psi4"] = Psi4Handle
  131 
  132     # Setup max iterations global variable...
  133     Psi4Util.UpdatePsi4OptionsParameters(Psi4Handle, {'GEOM_MAXITER': OptionsInfo["MaxIters"]})
  134     
  135     # Setup conversion factor for energy units...
  136     SetupEnergyConversionFactor(Psi4Handle)
  137     
  138     if OptionsInfo["SkipConformerGeneration"]:
  139         MiscUtil.PrintInfo("\nPerforming minimization without generation of conformers...")
  140     else:
  141         MiscUtil.PrintInfo("\nPerforming minimization with generation of conformers...")
  142 
  143     (MolCount, ValidMolCount, MinimizationFailedCount, WriteFailedCount) = [0] * 4
  144     for Mol in Mols:
  145         MolCount += 1
  146 
  147         if not CheckAndValidateMolecule(Mol, MolCount):
  148             continue
  149 
  150         # Setup 2D coordinates for SMILES input file...
  151         if OptionsInfo["SMILESInfileStatus"]:
  152             AllChem.Compute2DCoords(Mol)
  153         
  154         ValidMolCount += 1
  155         
  156         Mol, CalcStatus, ConfID, Energy = MinimizeMoleculeOrConformers(Psi4Handle, Mol, MolCount)
  157 
  158         if not CalcStatus:
  159             if not OptionsInfo["QuietMode"]:
  160                 MolName = RDKitUtil.GetMolName(Mol, MolCount)
  161                 MiscUtil.PrintWarning("Failed to calculate energy for molecule %s" % MolName)
  162             
  163             MinimizationFailedCount += 1
  164             continue
  165         
  166         Energy = "%.*f" % (OptionsInfo["Precision"], Energy)
  167         WriteStatus = WriteMolecule(Writer, Mol, MolCount, ConfID, Energy)
  168         
  169         if not WriteStatus:
  170             WriteFailedCount += 1
  171     
  172     return (MolCount, ValidMolCount, MinimizationFailedCount, WriteFailedCount)

  174 def ProcessMoleculesUsingMultipleProcesses(Mols, Writer):
  175     """Process and minimize molecules using multiprocessing."""
  176     
  177     if OptionsInfo["MPLevelConformersMode"]:
  178         return ProcessMoleculesUsingMultipleProcessesAtConformersLevel(Mols, Writer)
  179     elif OptionsInfo["MPLevelMoleculesMode"]:
  180         return ProcessMoleculesUsingMultipleProcessesAtMoleculesLevel(Mols, Writer)

  184 def ProcessMoleculesUsingMultipleProcessesAtMoleculesLevel(Mols, Writer):
  185     """Process and minimize molecules using multiprocessing at molecules level."""
  186 
  187     if OptionsInfo["SkipConformerGeneration"]:
  188         MiscUtil.PrintInfo("\nPerforming minimization without generation of conformers using multiprocessing at molecules level...")
  189     else:
  190         MiscUtil.PrintInfo("\nPerforming minimization with generation of conformers using multiprocessing at molecules level...")
  191 
  192     MPParams = OptionsInfo["MPParams"]
  193     
  194     # Setup data for initializing a worker process...
  195     InitializeWorkerProcessArgs = (MiscUtil.ObjectToBase64EncodedString(Options), MiscUtil.ObjectToBase64EncodedString(OptionsInfo))
  196     
  197     # Setup a encoded mols data iterable for a worker process...
  198     WorkerProcessDataIterable = RDKitUtil.GenerateBase64EncodedMolStrings(Mols)
  199     
  200     # Setup process pool along with data initialization for each process...
  201     if not OptionsInfo["QuietMode"]:
  202         MiscUtil.PrintInfo("\nConfiguring multiprocessing using %s method..." % ("mp.Pool.imap()" if re.match("^Lazy$", MPParams["InputDataMode"], re.I) else "mp.Pool.map()"))
  203         MiscUtil.PrintInfo("NumProcesses: %s; InputDataMode: %s; ChunkSize: %s\n" % (MPParams["NumProcesses"], MPParams["InputDataMode"], ("automatic" if MPParams["ChunkSize"] is None else MPParams["ChunkSize"])))
  204     
  205     ProcessPool = mp.Pool(MPParams["NumProcesses"], InitializeWorkerProcess, InitializeWorkerProcessArgs)
  206     
  207     # Start processing...
  208     if re.match("^Lazy$", MPParams["InputDataMode"], re.I):
  209         Results = ProcessPool.imap(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  210     elif re.match("^InMemory$", MPParams["InputDataMode"], re.I):
  211         Results = ProcessPool.map(WorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  212     else:
  213         MiscUtil.PrintError("The value, %s, specified for \"--inputDataMode\" is not supported." % (MPParams["InputDataMode"]))
  214     
  215     # Print out Psi4 version in the main process...
  216     MiscUtil.PrintInfo("\nInitializing Psi4...\n")
  217     Psi4Handle  = Psi4Util.InitializePsi4(PrintVersion = True, PrintHeader = False)
  218     OptionsInfo["psi4"] = Psi4Handle
  219     
  220     (MolCount, ValidMolCount, MinimizationFailedCount, WriteFailedCount) = [0] * 4
  221     for Result in Results:
  222         MolCount += 1
  223         MolIndex, EncodedMol, CalcStatus, ConfID, Energy = Result
  224         
  225         if EncodedMol is None:
  226             continue
  227         ValidMolCount += 1
  228 
  229         if not CalcStatus:
  230             MinimizationFailedCount += 1
  231             continue
  232             
  233         Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  234         
  235         Energy = "%.*f" % (OptionsInfo["Precision"], Energy)
  236         WriteStatus = WriteMolecule(Writer, Mol, MolCount, ConfID, Energy)
  237         
  238         if not WriteStatus:
  239             WriteFailedCount += 1
  240     
  241     return (MolCount, ValidMolCount, MinimizationFailedCount, WriteFailedCount)

  243 def InitializeWorkerProcess(*EncodedArgs):
  244     """Initialize data for a worker process."""
  245     
  246     global Options, OptionsInfo
  247 
  248     if not OptionsInfo["QuietMode"]:
  249         MiscUtil.PrintInfo("Starting process (PID: %s)..." % os.getpid())
  250     
  251     # Decode Options and OptionInfo...
  252     Options = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[0])
  253     OptionsInfo = MiscUtil.ObjectFromBase64EncodedString(EncodedArgs[1])
  254     
  255     # Psi4 is initialized in the worker process to avoid creation of redundant Psi4
  256     # output files for each process...
  257     OptionsInfo["Psi4Initialized"]  = False

  259 def WorkerProcess(EncodedMolInfo):
  260     """Process data for a worker process."""
  261 
  262     if not OptionsInfo["Psi4Initialized"]:
  263         InitializePsi4ForWorkerProcess()
  264     
  265     MolIndex, EncodedMol = EncodedMolInfo
  266 
  267     MolNum = MolIndex + 1
  268     CalcStatus = False
  269     ConfID = None
  270     Energy = None
  271     
  272     if EncodedMol is None:
  273         return [MolIndex, None, CalcStatus, ConfID, Energy]
  274 
  275     Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  276     
  277     if not CheckAndValidateMolecule(Mol, MolNum):
  278         return [MolIndex, None, CalcStatus, ConfID, Energy]
  279     
  280     # Setup 2D coordinates for SMILES input file...
  281     if OptionsInfo["SMILESInfileStatus"]:
  282         AllChem.Compute2DCoords(Mol)
  283         
  284     Mol, CalcStatus, ConfID, Energy = MinimizeMoleculeOrConformers(OptionsInfo["psi4"], Mol, MolNum)
  285 
  286     return [MolIndex, RDKitUtil.MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.MolProps | Chem.PropertyPickleOptions.PrivateProps), CalcStatus, ConfID, Energy]

  324 def ProcessConformersUsingMultipleProcesses(Mol, MolNum):
  325     """Generate conformers and minimize them using multiple processes. """
  326 
  327     # Add hydrogens...
  328     Mol = Chem.AddHs(Mol)
  329 
  330     # Setup conformers...
  331     ConfIDs = EmbedMolecule(Mol, MolNum)
  332     if not len(ConfIDs):
  333         if not OptionsInfo["QuietMode"]:
  334             MolName = RDKitUtil.GetMolName(Mol, MolNum)
  335             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
  336         return (Mol, False, None, None)
  337 
  338     MPParams = OptionsInfo["MPParams"]
  339     
  340     # Setup data for initializing a worker process...
  341     InitializeWorkerProcessArgs = (MiscUtil.ObjectToBase64EncodedString(Options), MiscUtil.ObjectToBase64EncodedString(OptionsInfo))
  342 
  343     # Setup a encoded mols data iterable for a worker process...
  344     MolIndex = MolNum - 1
  345     WorkerProcessDataIterable = RDKitUtil.GenerateBase64EncodedMolStringWithConfIDs(Mol, MolIndex, ConfIDs)
  346 
  347     # Setup process pool along with data initialization for each process...
  348     if not OptionsInfo["QuietMode"]:
  349         MiscUtil.PrintInfo("\nConfiguring multiprocessing using %s method..." % ("mp.Pool.imap()" if re.match("^Lazy$", MPParams["InputDataMode"], re.I) else "mp.Pool.map()"))
  350         MiscUtil.PrintInfo("NumProcesses: %s; InputDataMode: %s; ChunkSize: %s\n" % (MPParams["NumProcesses"], MPParams["InputDataMode"], ("automatic" if MPParams["ChunkSize"] is None else MPParams["ChunkSize"])))
  351     
  352     ProcessPool = mp.Pool(MPParams["NumProcesses"], InitializeConformerWorkerProcess, InitializeWorkerProcessArgs)
  353     
  354     # Start processing...
  355     if re.match("^Lazy$", MPParams["InputDataMode"], re.I):
  356         Results = ProcessPool.imap(ConformerWorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  357     elif re.match("^InMemory$", MPParams["InputDataMode"], re.I):
  358         Results = ProcessPool.map(ConformerWorkerProcess, WorkerProcessDataIterable, MPParams["ChunkSize"])
  359     else:
  360         MiscUtil.PrintError("The value, %s, specified for \"--inputDataMode\" is not supported." % (MPParams["InputDataMode"]))
  361     
  362     CalcEnergyMap = {}
  363     CalcFailedCount = 0
  364     for Result in Results:
  365         MolIndex, EncodedMol, CalcStatus, ConfID, Energy = Result
  366 
  367         if EncodedMol is None:
  368             CalcFailedCount += 1
  369             continue
  370         
  371         if not CalcStatus:
  372             CalcFailedCount += 1
  373             continue
  374         
  375         # Retrieve minimized atom positions...
  376         MinimizedMol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  377         AtomPositions = RDKitUtil.GetAtomPositions(MinimizedMol, ConfID = ConfID)
  378 
  379         # Update atom positions...
  380         RDKitUtil.SetAtomPositions(Mol, AtomPositions, ConfID = ConfID)
  381         
  382         CalcEnergyMap[ConfID] = Energy
  383     
  384     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
  385     MinEnergyConfID = SortedConfIDs[0]
  386     MinEnergy = CalcEnergyMap[MinEnergyConfID]
  387     
  388     if CalcFailedCount:
  389         return (Mol, False, None, None)
  390     
  391     return (Mol, True, MinEnergyConfID, MinEnergy)
  392

  409 def ConformerWorkerProcess(EncodedMolInfo):
  410     """Process data for a conformer worker process."""
  411 
  412     if not OptionsInfo["Psi4Initialized"]:
  413         InitializePsi4ForWorkerProcess()
  414     
  415     MolIndex, EncodedMol, ConfID = EncodedMolInfo
  416 
  417     MolNum = MolIndex + 1
  418     
  419     CalcStatus = False
  420     Energy = None
  421     
  422     if EncodedMol is None:
  423         return [MolIndex, None, CalcStatus, ConfID, Energy]
  424     
  425     Mol = RDKitUtil.MolFromBase64EncodedMolString(EncodedMol)
  426     MolName = RDKitUtil.GetMolName(Mol, MolNum)
  427     
  428     if not OptionsInfo["QuietMode"]:
  429         MiscUtil.PrintInfo("Processing molecule %s conformer ID %s..." % (MolName, ConfID))
  430     
  431     Status, ConvergeStatus = MinimizeMoleculeUsingForceField(Mol, MolNum, ConfID)
  432     if not Status:
  433         return [MolIndex, EncodedMol, CalcStatus, ConfID, Energy]
  434     
  435     if ConvergeStatus != 0:
  436         if not OptionsInfo["QuietMode"]:
  437             MolName = RDKitUtil.GetMolName(Mol, (MolIndex + 1))
  438             MiscUtil.PrintWarning("Minimization using forcefield failed to converge for molecule %s in %d steps. Try using higher value for \"maxIters\" in \"--confParams\" option...\n" % (MolName, OptionsInfo["ConfGenerationParams"]["MaxIters"]))
  439     
  440     # Perform Psi4 minimization...
  441     CalcStatus, Energy = MinimizeMoleculeUsingPsi4(OptionsInfo["psi4"], Mol, MolNum, ConfID)
  442     if not CalcStatus:
  443         if not OptionsInfo["QuietMode"]:
  444             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s\n" % (MolName))
  445             return [MolIndex, EncodedMol, False, ConfID, Energy]
  446 
  447     return [MolIndex, RDKitUtil.MolToBase64EncodedMolString(Mol, PropertyPickleFlags = Chem.PropertyPickleOptions.MolProps | Chem.PropertyPickleOptions.PrivateProps), CalcStatus, ConfID, Energy]

  473 def MinimizeMoleculeOrConformers(Psi4Handle, Mol, MolNum):
  474     """Minimize molecule or conformers."""
  475 
  476     ConfID = None
  477     if OptionsInfo["SkipConformerGeneration"]:
  478         Mol, CalcStatus, Energy = MinimizeMolecule(Psi4Handle, Mol, MolNum)
  479     else:
  480         Mol, CalcStatus, ConfID, Energy = GenerateAndMinimizeConformers(Psi4Handle, Mol, MolNum)
  481 
  482     return (Mol, CalcStatus, ConfID, Energy)

  484 def GenerateAndMinimizeConformers(Psi4Handle, Mol, MolNum = None):
  485     """Generate and minimize conformers for a molecule to get the lowest energy conformer."""
  486 
  487     # Add hydrogens..
  488     Mol = Chem.AddHs(Mol)
  489     
  490     MolName = RDKitUtil.GetMolName(Mol, MolNum)
  491 
  492     # Setup conformers...
  493     ConfIDs = EmbedMolecule(Mol, MolNum)
  494     if not len(ConfIDs):
  495         if not OptionsInfo["QuietMode"]:
  496             MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s: Embedding failed...\n" % MolName)
  497         return (Mol, False, None, None)
  498     
  499     # Minimize conformers...
  500     CalcEnergyMap = {}
  501     for ConfID in ConfIDs:
  502         if not OptionsInfo["QuietMode"]:
  503             MiscUtil.PrintInfo("Processing molecule %s conformer ID %s..." % (MolName, ConfID))
  504             
  505         # Perform forcefield minimization...
  506         Status, ConvergeStatus = MinimizeMoleculeUsingForceField(Mol, MolNum, ConfID)
  507         if not Status:
  508             return (Mol, False, None, None)
  509         
  510         if ConvergeStatus != 0:
  511             if not OptionsInfo["QuietMode"]:
  512                 MiscUtil.PrintWarning("Minimization using forcefield failed to converge for molecule %s in %d steps. Try using higher value for \"maxIters\" in \"--confParams\" option...\n" % (MolName, OptionsInfo["ConfGenerationParams"]["MaxIters"]))
  513         
  514         # Perform Psi4 minimization...
  515         CalcStatus, Energy = MinimizeMoleculeUsingPsi4(Psi4Handle, Mol, MolNum, ConfID)
  516         if not CalcStatus:
  517             if not OptionsInfo["QuietMode"]:
  518                 MiscUtil.PrintWarning("Minimization couldn't be performed for molecule %s\n" % (MolName))
  519             return (Mol, False, None, None)
  520 
  521         CalcEnergyMap[ConfID] = Energy
  522 
  523     # Select the lowest energy conformer..
  524     SortedConfIDs = sorted(ConfIDs, key = lambda ConfID: CalcEnergyMap[ConfID])
  525     MinEnergyConfID = SortedConfIDs[0]
  526     Energy = CalcEnergyMap[MinEnergyConfID]
  527 
  528     return (Mol, True, MinEnergyConfID, Energy)
  529 
  530 def MinimizeMolecule(Psi4Handle, Mol, MolNum):
  531     """Minimize molecule."""
  532 
  533     # Skip conformation generation and forcefield minimization...
  534     CalcStatus, Energy = MinimizeMoleculeUsingPsi4(Psi4Handle, Mol, MolNum)
  535     
  536     return (Mol, CalcStatus, Energy)

  538 def MinimizeMoleculeUsingPsi4(Psi4Handle, Mol, MolNum, ConfID = -1):
  539     """Minimize molecule using Psi4."""
  540 
  541     # Setup a Psi4Mol...
  542     Psi4Mol = SetupPsi4Mol(Psi4Handle, Mol, MolNum, ConfID)
  543     if Psi4Mol is None:
  544         return (False, None)
  545         
  546     #  Setup reference wave function...
  547     Reference = SetupReferenceWavefunction(Mol)
  548     Psi4Handle.set_options({'Reference': Reference})
  549     
  550     # Setup method name and basis set...
  551     MethodName, BasisSet = SetupMethodNameAndBasisSet(Mol)
  552 
  553     # Optimize geometry...
  554     Status, Energy, WaveFunction = Psi4Util.PerformGeometryOptimization(Psi4Handle, Psi4Mol, MethodName, BasisSet, ReturnWaveFunction = True, Quiet = OptionsInfo["QuietMode"])
  555     
  556     if not Status:
  557         PerformPsi4Cleanup(Psi4Handle)
  558         return (False, None)
  559 
  560     # Update atom positions...
  561     AtomPositions = Psi4Util.GetAtomPositions(Psi4Handle, WaveFunction, InAngstroms = True)
  562     RDKitUtil.SetAtomPositions(Mol, AtomPositions, ConfID = ConfID)
  563 
  564     # Convert energy units...
  565     if OptionsInfo["ApplyEnergyConversionFactor"]:
  566         Energy = Energy * OptionsInfo["EnergyConversionFactor"]
  567     
  568     # Clean up
  569     PerformPsi4Cleanup(Psi4Handle)
  570     
  571     return (True, Energy)

  573 def MinimizeMoleculeUsingForceField(Mol, MolNum, ConfID = -1):
  574     """Minimize molecule using forcefield available in RDKit."""
  575     
  576     try:
  577         if OptionsInfo["ConfGenerationParams"]["UseUFF"]:
  578             ConvergeStatus = AllChem.UFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["ConfGenerationParams"]["MaxIters"])
  579         elif OptionsInfo["ConfGenerationParams"]["UseMMFF"]:
  580             ConvergeStatus = AllChem.MMFFOptimizeMolecule(Mol, confId = ConfID, maxIters = OptionsInfo["ConfGenerationParams"]["MaxIters"], mmffVariant = OptionsInfo["ConfGenerationParams"]["MMFFVariant"])
  581         else:
  582             MiscUtil.PrintError("Minimization couldn't be performed: Specified forcefield, %s, is not supported" % OptionsInfo["ConfGenerationParams"]["ForceField"])
  583     except (ValueError, RuntimeError, Chem.rdchem.KekulizeException) as ErrMsg:
  584         if not OptionsInfo["QuietMode"]:
  585             MolName = RDKitUtil.GetMolName(Mol, MolNum)
  586             MiscUtil.PrintWarning("Minimization using forcefield couldn't be performed for molecule %s:\n%s\n" % (MolName, ErrMsg))
  587         return (False, None)
  588     
  589     return (True, ConvergeStatus)

  652 def CheckAndValidateMolecule(Mol, MolCount = None):

  711 def SetupEnergyConversionFactor(Psi4Handle):

  716     ApplyConversionFactor = True
